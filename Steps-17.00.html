·         Drag and drop the OLE DB Source – Double click and add the Source connection manager and Data access mode select SQL Command – Copy the below code and paste in the SQL command text box.  And select OK.

with Cte 
as
(
select [Incident Number w/year],
[Offense Status], [Year of Incident], [Month of Occurence],
[Day1 of the Year],
ROW_NUMBER()OVER(PARTITION BY [Offense Status], [Year of Incident], [Month of Occurence], [Day1 of the Year] order by [Day1 of the Year], [Month of Occurence],[Year of Incident] ) as RoWnum
from DallasCrimeData_Staging..DallasCrimeStaging
group by [Offense Status], [Year of Incident], [Month of Occurence], [Day1 of the Year],[Incident Number w/year]
)
select [Incident Number w/year],[Suspended], [Clear by Exceptional Arrest], [Clear by Arrest], [Open], [Closed/Cleared], [Year of Incident], [Month of Occurence], [Day1 of the Year]
from
( select [Incident Number w/year], [Offense Status], [Year of Incident], [Month of Occurence], [Day1 of the Year], RoWnum 
from cte) p
pivot
(
Count(RoWnum) for [Offense Status] in ([Suspended], [Clear by Exceptional Arrest], [Clear by Arrest], [Open], [Closed/Cleared])
) as PVT
order by [Year of Incident], [Month of Occurence], [Day1 of the Year], [Incident Number w/year]
